# main.py from fastapi import FastAPI, Request from fastapi.responses import HTMLResponse from pydantic import BaseModel app = FastAPI() # "Banco de dados" em memória usuarios = [] # Modelo de dados class Usuario(BaseModel): nome: str idade: int # --- 1. GET "/" -> retorna HTML --- @app.get("/", response_class=HTMLResponse) async def home(): return """ Requests

Requests

Respostas

""" # --- 2. POST /users -> adicionar usuário --- @app.post("/users") async def adicionar_usuario(request: Request): form = await request.form() nome = form.get("nome") idade = form.get("idade") usuario = {"nome": nome, "idade": int(idade)} usuarios.append(usuario) return usuario # --- 3. GET /users -> listar ou pegar por índice --- @app.get("/users") async def listar_usuarios(index: int | None = None): if index is not None: if 0 <= index < len(usuarios): return usuarios[index] return {"erro": "Índice inválido"} return usuarios # --- 4. DELETE /users -> limpar lista --- @app.delete("/users") async def deletar_usuarios(): usuarios.clear() return {"mensagem": "Lista limpa"}
